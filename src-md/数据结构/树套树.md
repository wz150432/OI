# 树套树

这个数据结构就像他的名字一样，一个数据结构套另一个数据结构，一般外套树状数组或者线段树。内套平衡树或者线段树。

下面是线段树套平衡树的模板，等有时间再拆分。

```cpp
#include <bits/stdc++.h>

using namespace std;

const int N = 2e6 + 10;
const int INF = 2147483647;

int n, m, idx;
int arr[N];

struct node {
    int p, v;
    int sz, s[2];
    
    void init(int _p, int _v) {
        p = _p;
        v = _v;
        sz = 1;
    }
}tr[N];

int L[N], R[N], S[N];

void pushup(int u) {
    tr[u].sz = tr[tr[u].s[0]].sz + tr[tr[u].s[1]].sz + 1;
}

void rotate(int x) {
    int y = tr[x].p, z = tr[y].p;
    int k = tr[y].s[1] == x;
    tr[z].s[tr[z].s[1] == y] = x;
    tr[x].p = z;
    tr[y].s[k] = tr[x].s[k ^ 1];
    tr[tr[x].s[k ^ 1]].p = y;
    tr[x].s[k ^ 1] = y;
    tr[y].p = x;
    pushup(y), pushup(x);
}

int find_k(int u, int k) {
    int res = 0;
    while (u) {
        if (tr[u].v < k) {
            res += tr[tr[u].s[0]].sz + 1;
            u = tr[u].s[1];
        } else u = tr[u].s[0];
    }
    return res;
}

int find_pre(int root, int v) {
    int u = root, res = -INF;
    while (u) {
        if (tr[u].v < v) {
            res = max(tr[u].v, res);
            u = tr[u].s[1];
        } else u = tr[u].s[0];
    }
    return res;
}

int find_suf(int root, int v) {
    int u = root, res = INF;
    while (u) {
        if (tr[u].v > v) {
            res = min(res, tr[u].v);
            u = tr[u].s[0];
        } else u = tr[u].s[1];
    }
    return res;
}

void splay(int &root, int x, int k) {
     while (tr[x].p != k) {
         int y = tr[x].p, z = tr[y].p;
         if (z != k) {
             if ((tr[z].s[1] == y) ^ (tr[y].s[1] == x)) rotate(x);
             else rotate(y);
         }
         rotate(x);
     }
     if (!k) root = x;
}

void insert(int &root, int x) {
    int u = root, p = 0;
    while (u) {
        p = u;
        u = tr[u].s[x > tr[u].v];
    }
    u = ++ idx;
    if (p) tr[p].s[x > tr[p].v] = u;
    tr[u].init(p, x);
    splay(root, u, 0);
}

void build(int u, int l, int r) {
    L[u] = l, R[u] = r;
    insert(S[u], -INF), insert(S[u], INF);
    for (int i = l; i <= r; ++ i) insert(S[u], arr[i]);
    if (l != r) {
        int mid = (l + r) >> 1;
        build(u << 1, l, mid);
        build(u << 1 | 1, mid + 1, r);
    }
}

void update(int &root, int x, int y) {
    int u = root;
    while (u) {
        if (tr[u].v == x) break;
        if (tr[u].v < x) u = tr[u].s[1];
        else u = tr[u].s[0];
    }
    splay(root, u, 0);
    int l = tr[u].s[0], r = tr[u].s[1];
    while (tr[l].s[1]) l = tr[l].s[1];
    while (tr[r].s[0]) r = tr[r].s[0];
    splay(root, l, 0), splay(root, r, l);
    tr[r].s[0] = 0;
    pushup(r), pushup(l);
    insert(root, y);
}

void modify(int u, int k, int c) {
    update(S[u], arr[k], c);
    if (L[u] != R[u]) {
        int mid = (L[u] + R[u]) >> 1;
        if (k <= mid) modify(u << 1, k, c);
        else modify(u << 1 | 1, k, c);
    }
}

int query_pre(int u, int l, int r, int k) {
    if (L[u] >= l && R[u] <= r) return find_pre(S[u], k);
    int mid = (L[u] + R[u]) >> 1, res = -INF;
    if (l <= mid) res = max(res, query_pre(u << 1, l, r, k));
    if (r > mid) res = max(res, query_pre(u << 1 | 1, l, r, k));
    return res;
}

int query_suf(int u, int l, int r, int k) {
    if (L[u] >= l && R[u] <= r) return find_suf(S[u], k);
    int mid = (L[u] + R[u]) >> 1, res = INF;
    if (l <= mid) res = min(res, query_suf(u << 1, l, r, k));
    if (r > mid) res = min(res, query_suf(u << 1 | 1, l, r, k));
    return res;
}

int query(int u, int l, int r, int k) {
    if (L[u] >= l && R[u] <= r) return find_k(S[u], k) - 1;
    int mid = (L[u] + R[u]) >> 1, res = 0;
    if (l <= mid) res += query(u << 1, l, r, k);
    if (r > mid) res += query(u << 1 | 1, l, r, k);
    return res;
}

int main() {
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; ++ i) scanf("%d", &arr[i]);
    
    build(1, 1, n);
    while (m -- ) {
        int op, l, r, x;
        scanf("%d", &op);
        if (op == 1) {
            scanf("%d%d%d", &l, &r, &x);
            printf("%d\n", query(1, l, r, x) + 1);
        } else if (op == 2) {
            scanf("%d%d%d", &l, &r, &x);
            int a = 0, b = 1e8;
            while (a < b) {
                int mid = (a + b + 1) >> 1;
                if (query(1, l, r, mid) + 1 <= x) a = mid;
                else b = mid - 1;
            }
            printf("%d\n", b);
        } else if (op == 3) {
            scanf("%d%d", &l, &x);
            modify(1, l, x);
            arr[l] = x; 
        } else if (op == 4) {
            scanf("%d%d%d", &l, &r, &x);
            printf("%d\n", query_pre(1, l, r, x));
        } else {
            scanf("%d%d%d", &l, &r, &x);
            printf("%d\n", query_suf(1, l, r, x));
        }
    }
    
    return 0;
}
```
